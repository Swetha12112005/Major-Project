        function calculatePrediction(e) {
            e.preventDefault();

            // Get Values
            const name = document.getElementById('fullName').value;
            const cgpa = parseFloat(document.getElementById('cgpa').value);
            const backlogs = parseInt(document.getElementById('backlogs').value) || 0;
            const projects = parseInt(document.getElementById('projects').value) || 0;
            const aptitude = parseInt(document.getElementById('aptitude').value) || 0;
            const communication = parseInt(document.getElementById('communication').value) || 0;
            const internshipCount = parseInt(document.getElementById('internship_count').value) || 0;
            const internshipType = document.getElementById('internship_type').value;
            const dept = document.getElementById('dept').value;
            const skill = document.getElementById('skill').value;

            // 1. Base Score Calculation
            let score = 0;

            // CGPA Weight (Max 40 points)
            score += (cgpa * 4);

            // Aptitude Weight (Max 15 points)
            score += (aptitude * 0.15);

            // Projects Weight (Max 10 points)
            score += (projects * 2);

            // Internship Weight (Max 20 points)
            let internshipScore = 0;
            if (internshipCount > 0) {
                internshipScore += (internshipCount * 5); // Base points for count
                if (internshipType === 'mnc_product') internshipScore += 10;
                else if (internshipType === 'startup') internshipScore += 5;
                else if (internshipType === 'research') internshipScore += 3;
            }
            score += internshipScore;

            // Communication Weight (Max 10 points)
            score += (communication * 1);

            // Department Bonus
            if (dept === 'cse_ai' || dept === 'cse_ml' || dept === 'cse_ds' || dept === 'cyber') score += 2;
            else if (dept === 'mech' || dept === 'civil') score -= 2; // Harder market usually

            // Penalty
            score -= (backlogs * 4);

            // Clamp
            let finalScore = Math.floor(score);
            if (finalScore > 99) finalScore = 99;
            if (finalScore < 15) finalScore = 15;

            // Update UI
            document.getElementById('resultName').innerText = `Candidate: ${name}`;
            document.getElementById('predictionForm').style.display = 'none';
            document.getElementById('resultSection').style.display = 'block';
            
            animateValue("scoreValue", 0, finalScore, 2000);
            
            // Target Text Logic
            let targetText = "";
            if (finalScore > 85) targetText = "High Probability: Product-Based Companies (FAANG, Unicorns)";
            else if (finalScore > 70) targetText = "Good Probability: Service-Based MNCs & Growing Startups";
            else if (backlogs > 0) targetText = "Priority: Clear Backlogs Immediately to Boost Profile";
            else targetText = "Needs Improvement: Focus on Skill Building & Projects";
            
            document.getElementById('companyTarget').innerText = targetText;

            // Bar Chart Calculation & Animation
            let acadScore = (cgpa * 10) - (backlogs * 10);
            if(acadScore < 0) acadScore = 20;
            
            // Tech Score (Aptitude + Projects + Skill choice bonus)
            let techScore = aptitude + (projects * 5);
            if(techScore > 100) techScore = 100;

            // Soft Skill Score (Communication + Internship Type)
            let softScore = (communication * 8) + (internshipType === 'mnc_product' ? 15 : 0);
            if(softScore > 100) softScore = 100;

            setTimeout(() => {
                document.getElementById('fill1').style.width = acadScore + "%";
                document.getElementById('bar1Val').innerText = acadScore + "%";
                
                document.getElementById('fill2').style.width = techScore + "%";
                document.getElementById('bar2Val').innerText = techScore + "%";
                
                document.getElementById('fill3').style.width = softScore + "%";
                document.getElementById('bar3Val').innerText = softScore + "%";

                document.getElementById('progressCircle').style.background = `conic-gradient(var(--primary) ${finalScore * 3.6}deg, #e2e8f0 0deg)`;
            }, 500);

            showToast("Analysis Complete");
        }

        function animateValue(id, start, end, duration) {
            if (start === end) return;
            let range = end - start;
            let current = start;
            let increment = end > start ? 1 : -1;
            let stepTime = Math.abs(Math.floor(duration / range));
            let obj = document.getElementById(id);
            let timer = setInterval(function() {
                current += increment;
                obj.innerHTML = current + "%";
                if (current == end) { clearInterval(timer); }
            }, stepTime);
        }
